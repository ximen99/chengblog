How about we do some clustering see how close are different types of cuisine related just based on the ingredients they use. In order to do that we'll first use TF-IDF transformation to normalize ingredients count and use Truncated SVD for dimensionality reduction so that we can visulize and use k-means to cluster them.

Embedding and MLP to Predict Cuisine Type¶

Till now we used bag of word, Tfidf in feature engineering to analyze the relationship among different cuisine types. It works pretty well. However if we want to catch something more subtle besides word frequency, embedding seems to be the most powerful and convenient technique for job. For illustration purpose, I'll build a simple neural network to predict the cuisine type based on the ingredients using embedding.

def prepare_data(test_size=0.2):
    """Prepare train test dataset
    test_size -- proportion of test data 
    """
    # create label encoder
    le = LabelEncoder().fit(ingredients_selected)

    # deleted ingredients appeared less than twice
    X_raw = data.ingredients.apply(lambda x: [i for i in x if i in ingredients_selected] )

    # label encode ingredients
    X_le = X_raw.apply(le.transform)

    # one hot encode cuisine type
    oe = OneHotEncoder(sparse = False)
    y = oe.fit_transform(data[['cuisine']])

    # pad ingredients
    max_len = 20
    X = pad_sequences(X_le,maxlen=max_len)

    # train test split
    return train_test_split(X, y, test_size=test_size, random_state=42)

X_train, X_test, y_train, y_test = prepare_data(test_size = 0.2)
In [239]:
# create MLP model
def baseline_model():
    # create model
    model = Sequential([
        layers.Embedding(input_dim=4048, output_dim=8, mask_zero=True, input_length = max_len, name = 'Embedding'),
        layers.Flatten(),
        layers.Dropout(0.2),
        layers.Dense(20, activation='softmax', name = 'Dense_output')
    ])

    # Compile model
    model.compile(loss='categorical_crossentropy', optimizer='adam', metrics=['accuracy'])
    return model

def train_model():
    # build model
    model = baseline_model()
    # log loss and metrics to tensorboard
    logdir = os.path.join("logs", datetime.datetime.now().strftime("%Y%m%d-%H%M%S"))
    tensorboard_callback = tf.keras.callbacks.TensorBoard(logdir, histogram_freq=1)
    # train model
    model.fit(x=X_train, 
              y=y_train, 
              epochs=20, 
              validation_data=(X_test, y_test),
              callbacks=[tensorboard_callback])
    
    return model
